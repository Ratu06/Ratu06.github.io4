📖 Rangkuman Materi (Slide 1–16)

  • 🔹 Polimorfisme : objek child dapat diperlakukan sebagai tipe parent (runtime via overriding).
  • 🔹 is & as : pengecekan tipe & casting aman.
  • 🔹 Abstraksi : menyembunyikan detail, fokus ke "apa". Dart gunakan abstract class dan interface class.
  • 🔹 Abstract class : tidak bisa diinstansiasi, bisa punya method konkret & abstrak.
  • 🔹 Interface : kontrak penuh, class mengimplementasi dengan implements (multiple).
  • 🔹 extends vs implements : "is-a" vs "can-do".
  • 🔹 Static members : milik class, diakses via ClassName. Berguna untuk utility.
  • 🔹 Manfaat : fleksibel, mudah diperluas, kode reusable, maintainability tinggi.

🎭 Polimorfisme dengan Inheritance

Referensi parent bisa menampung objek child. Method yang dipanggil sesuai objek asli (runtime).
// Bentuk, Lingkaran, Persegi
abstract class Bentuk {
  void gambar();
}

class Lingkaran extends Bentuk {
  @override
  void gambar() => print('Menggambar lingkaran ⚪');
}

class Persegi extends Bentuk {
  @override
  void gambar() => print('Menggambar persegi 🟩');
}

void main() {
  Bentuk bentuk1 = Lingkaran();
  Bentuk bentuk2 = Persegi();
  bentuk1.gambar();  // lingkaran
  bentuk2.gambar();  // persegi
}

🔍 Operator is dan as

Cek tipe objek dan konversi aman.
class Kendaraan {}
class Mobil extends Kendaraan {
  void klakson() => print('Tin! Tin!');
}

void main() {
  Kendaraan k = Mobil();
  if (k is Mobil) {
    (k as Mobil).klakson();
  }
  // safe casting with as?
  (k as Mobil?)?.klakson();
}

🐾 Praktik 1 – Polimorfisme dengan List Hewan

abstract class Hewan {
  void bersuara();
}
class Kucing extends Hewan {
  @override void bersuara() => print('Meong 🐱');
}

  @override void bersuara() => print('Guk guk 🐶');

🧱 Abstract Class & Method

abstract class Kendaraan {
  String? merk;
  void start(); // abstract
  void stop() => print('Kendaraan berhenti');
}
class Mobil extends Kendaraan {
  @override
  void start() => print('Mobil ${merk ?? 'tanpa merk'} menyala');
}
void main() {
  Mobil avanza = Mobil();
  avanza.merk = 'Toyota';
  avanza.start();
  avanza.stop();
}

🛸 Interface (implements)

interface class Flyable {
  void terbang() => print('Terbang...');
}
class Bird implements Flyable {
  @override void terbang() => print('Burung terbang tinggi');
}
class Plane implements Flyable {
  @override void terbang() => print('Pesawat terbang cepat');
}
void main() {
  Flyable f1 = Bird();
  Flyable f2 = Plane();
  f1.terbang();
  f2.terbang();
}

💳 Praktik 2 – Sistem Pembayaran (Interface)

}
class KartuKredit implements MetodePembayaran {
  @override void bayar(double jumlah) => print('💳 Bayar $jumlah pakai Kartu Kredit');
}
class EWallet implements MetodePembayaran {
  @override void bayar(double jumlah) => print('📱 Bayar $jumlah via E-Wallet');

  @override void bayar(double jumlah) => print('💵 Bayar $jumlah dengan Tunai');

  for (var m in methods) {
    m.bayar(75000);
  }
}

📱 Tablet – Multiple Inheritance dengan implements/mixins

4. Class Tablet mewarisi kemampuan Komputer, Kamera, Telepon (multiple inheritance disimulasikan dengan implements).
class Komputer {
  void komputasi() => print('💻 Menjalankan sistem operasi');
}
class Kamera {
  void ambilFoto() => print('📸 Mengambil foto');
}
class Telepon {
  void telpon(String nomor) => print('📞 Menelpon $nomor');

  Tablet myTab = Tablet();
  myTab.komputasi();
  myTab.ambilFoto();
  myTab.telpon('08123456789');
}

🧮 MathUtils – Static Methods (Faktorial, Prima, Random, Rata2, Median)

import 'dart:math';

class MathUtils {
  static int factorial(int n) {
    if (n < 0) throw ArgumentError('Harus >=0');
    return n <= 1 ? 1 : n * factorial(n - 1);
  }
  static bool isPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i <= sqrt(n); i++) {
      if (n % i == 0) return false;
    }
    return true;
  }
  static double roundToDecimals(double value, int decimals) {
    double mod = pow(10, decimals).toDouble();
    return (value * mod).round() / mod;
  }
  static int randomInt(int min, int max) => min + Random().nextInt(max - min + 1);
  static double randomDouble(double min, double max) => min + Random().nextDouble() * (max - min);
  static double average(List<num> numbers) => numbers.isEmpty ? 0 : numbers.reduce((a,b)=>a+b) / numbers.length;
  static double median(List<num> numbers) {
    if (numbers.isEmpty) return 0;
    var sorted = List<num>.from(numbers)..sort();
    int mid = sorted.length ~/ 2;
    if (sorted.length.isEven) {
      return (sorted[mid - 1] + sorted[mid]) / 2;
    } else {
      return sorted[mid].toDouble();
    }
  }
}

void main() {
  print('Faktorial 5: ${MathUtils.factorial(5)}');
  print('17 prima? ${MathUtils.isPrime(17)}');
  print('3.14159 2 desimal: ${MathUtils.roundToDecimals(3.14159, 2)}');
  print('Random int 1-10: ${MathUtils.randomInt(1,10)}');
  print('Random double 0-1: ${MathUtils.randomDouble(0,1).toStringAsFixed(4)}');
  print('Rata2 [2,4,6]: ${MathUtils.average([2,4,6])}');
  print('Median [3,1,5,7]: ${MathUtils.median([3,1,5,7])}');
}

🎵 Challenge: MediaPlayer + Playable/Pausable

Abstract MediaPlayer, interface Playable & Pausable, class AudioPlayer & VideoPlayer.
abstract class MediaPlayer {
  void info() => print('Media Player aktif');
}
interface class Playable {
  void play();
}
interface class Pausable {
  void pause();
}
class AudioPlayer implements MediaPlayer, Playable, Pausable {
  @override void play() => print('🎵 Memutar audio...');
  @override void pause() => print('⏸ Audio dijeda');
}
class VideoPlayer implements MediaPlayer, Playable, Pausable {
  @override void play() => print('🎬 Memutar video...');
  @override void pause() => print('⏸ Video dijeda');
}
void main() {
  List<Playable> players = [AudioPlayer(), VideoPlayer()];
  for (var p in players) {
    p.play();
    if (p is Pausable) p.pause();
    print('---');
  }
}
